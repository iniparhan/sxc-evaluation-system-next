generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
  directUrl = env("DIRECT_URL")
}

model members {
  id               Int      @id @default(autoincrement())
  name             String   @db.VarChar(100)
  email            String   @unique @db.VarChar(150)
  password         String   @db.VarChar(255)

  role_id          Int?
  role             roles?   @relation(fields: [role_id], references: [id])

  division_id      Int?
  sub_division_id  Int?

  remember_token   String?  @db.VarChar(100)

  created_at       DateTime @default(now())
  updated_at       DateTime @default(now())

  // relations
  division         divisions?     @relation(fields: [division_id], references: [id])
  sub_division     sub_divisions? @relation(fields: [sub_division_id], references: [id])

  evaluator        evaluations[]  @relation("evaluator")
  evaluatee        evaluations[]  @relation("evaluatee")

  evaluation_results evaluation_results[]
  audit_logs        audit_logs[]

  // CHECK constraint: members_role_structure_check
  // (division_id IS NULL AND sub_division_id IS NULL)
  // OR (division_id IS NOT NULL AND sub_division_id IS NULL)
  // OR (division_id IS NOT NULL)

  @@map("members")
}

model roles {
  id    Int    @id @default(autoincrement())
  name  String @unique @db.VarChar(50)

  members          members[]

  evaluator_policies evaluation_policies[] @relation("evaluator_role")
  evaluatee_policies evaluation_policies[] @relation("evaluatee_role")

  @@map("roles")
}

model divisions {
  id    Int    @id @default(autoincrement())
  name  String @unique @db.VarChar(100)

  sub_divisions sub_divisions[]
  members       members[]
  kpis          kpis[]

  @@map("divisions")
}

model sub_divisions {
  id          Int    @id @default(autoincrement())
  name        String @db.VarChar(100)

  division_id Int

  division    divisions @relation(fields: [division_id], references: [id], onDelete: Cascade)
  members     members[]

  @@map("sub_divisions")
}

model evaluation_policies {
  id                  Int     @id @default(autoincrement())

  evaluator_role_id   Int
  evaluatee_role_id   Int

  division_scope      DivisionScope

  priority            Int     @default(1)
  is_active           Boolean @default(true)

  created_at          DateTime @default(now())

  evaluator_role roles @relation("evaluator_role", fields: [evaluator_role_id], references: [id])
  evaluatee_role roles @relation("evaluatee_role", fields: [evaluatee_role_id], references: [id])

  @@map("evaluation_policies")
}

model evaluation_periods {
  id         Int      @id @default(autoincrement())
  name       String   @db.VarChar(100)
  quartal    Int
  is_active  Boolean  @default(true)
  created_at DateTime @default(now())

  evaluations        evaluations[]
  evaluation_results evaluation_results[]

  @@map("evaluation_periods")
}

model kpis {
  id          Int     @id @default(autoincrement())

  name        String  @db.VarChar(150)
  code        String  @unique @db.VarChar(20)
  indicator   String  @db.Text

  division_id Int?

  weight      Int     @default(1)
  max_score   Int     @default(6)

  version     Int     @default(1)
  is_active   Boolean @default(true)

  created_at  DateTime @default(now())
  updated_at  DateTime @default(now())

  division    divisions? @relation(fields: [division_id], references: [id])

  indicators  kpi_indicators[]
  scores      evaluation_scores[]

  @@map("kpis")
}

model kpi_indicators {
  id          Int     @id @default(autoincrement())
  kpi_id      Int
  description String

  created_at  DateTime @default(now())

  kpi         kpis @relation(fields: [kpi_id], references: [id], onDelete: Cascade)

  @@map("kpi_indicators")
}

model evaluations {
  id            Int      @id @default(autoincrement())

  evaluator_id  Int
  evaluatee_id  Int

  period_id     Int?

  submitted_at  DateTime?

  created_at    DateTime @default(now())

  evaluator members @relation("evaluator", fields: [evaluator_id], references: [id])
  evaluatee members @relation("evaluatee", fields: [evaluatee_id], references: [id])

  period     evaluation_periods? @relation(fields: [period_id], references: [id])

  scores     evaluation_scores[]

  // CHECK constraint: no_self_evaluation - evaluator_id <> evaluatee_id

  @@unique([evaluator_id, evaluatee_id, period_id])
  @@map("evaluations")
}

model evaluation_scores {
  id             Int     @id @default(autoincrement())
  evaluation_id  Int
  kpi_id         Int

  score          Int
  notes          String?

  created_at     DateTime @default(now())
  updated_at     DateTime @default(now())

  evaluation evaluations @relation(fields: [evaluation_id], references: [id], onDelete: Cascade)
  kpi        kpis        @relation(fields: [kpi_id], references: [id])

  // CHECK constraint: score_range - score BETWEEN 1 AND 6

  @@map("evaluation_scores")
}

model evaluation_results {
  id            Int     @id @default(autoincrement())

  member_id     Int
  period_id     Int?

  total_score   Float   @default(0)
  average_score Float   @default(0)
  ranking       Int?

  created_at    DateTime @default(now())

  member members @relation(fields: [member_id], references: [id])
  period evaluation_periods? @relation(fields: [period_id], references: [id])

  @@map("evaluation_results")
}

model audit_logs {
  id          Int     @id @default(autoincrement())

  actor_id    Int?

  action      String  @db.VarChar(100)
  entity_type String  @db.VarChar(50)
  entity_id   Int

  before_data Json?
  after_data  Json?

  created_at  DateTime @default(now())

  actor members? @relation(fields: [actor_id], references: [id])

  @@map("audit_logs")
}

enum DivisionScope {
  GLOBAL
  SAME_DIVISION
  SAME_SUBDIVISION
}
